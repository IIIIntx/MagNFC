13/**
14 * @defgroup MODS_NSS_STORAGE_DFT Diversity Settings
15 * @ingroup MODS_NSS_STORAGE
16 *
17 * The application can adapt the storage module to better fit the different application scenarios through the use of
18 * diversity settings in the form of defines below. Sensible defaults are chosen. To override the default settings,
19 * place the defines with their desired values in the application app_sel.h header file. The compiler will pick up your
20 * defines before parsing this file.
21 *
22 * Additional notes regarding some flags:
23 * - By default, the assigned EEPROM region takes up 2KB and is located just below the read-only EEPROM rows. This
24 * storage space can be moved and resized by adapting #STORAGE_EEPROM_FIRST_ROW and #STORAGE_EEPROM_LAST_ROW.
25 * - By default, the remaining FLASH memory after programming is used for data storage: the available free
26 * FLASH space is automatically determined. It can also be manually assigned: move and resize the storage space in
27 * FLASH by adapting #STORAGE_FLASH_FIRST_PAGE and #STORAGE_FLASH_LAST_PAGE.
28 * - Data is stored decompressed in EEPROM; a chance is given to the application to compress the data before it is moved
29 * to FLASH. For this, define both #STORAGE_COMPRESS_CB and #STORAGE_DECOMPRESS_CB.
30 * - Flash operations should be used sparingly. The amount of data transferred as one block from EEPROM to FLASH can be
31 * controlled using #STORAGE_BLOCK_SIZE_IN_SAMPLES. Too little data puts unnecessary strain on the battery and reduces
32 * compression abilities; too much data risks a portion of the FLASH to remain not utilized when a compressed block of
33 * samples no longer fits.
34 * - The storage module requires the use of at least one general purpose register. This cannot be disabled.
35 *
36 * These flags may be overridden:
37 * - #STORAGE_FIRST_ALON_REGISTER
38 * - #STORAGE_SAMPLE_ALON_CACHE_COUNT
39 * - #STORAGE_EEPROM_FIRST_ROW
40 * - #STORAGE_EEPROM_LAST_ROW
41 * - #STORAGE_FLASH_FIRST_PAGE
42 * - #STORAGE_FLASH_LAST_PAGE
43 * - #STORAGE_TYPE
44 * - #STORAGE_BITSIZE
45 * - #STORAGE_SIGNED
46 * - #STORAGE_WORKAREA
47 * - #STORAGE_WRITE_RECOVERY_EVERY_X_SAMPLES
48 * - #STORAGE_BLOCK_SIZE_IN_SAMPLES
49 * - #STORAGE_REDUCE_RECOVERY_WRITES
50 * - #STORAGE_COMPRESS_CB
51 * - #STORAGE_DECOMPRESS_CB
52 *
53 * These defines are fixed or derived from the above flags and may not be defined or redefined in an application:
54 * - #STORAGE_EEPROM_ROW_COUNT
55 * - #STORAGE_EEPROM_SIZE
56 * - #STORAGE_MAX_SAMPLE_ALON_CACHE_COUNT
57 * - #STORAGE_WORKAREA_SELF_DEFINED
58 * - #STORAGE_WORKAREA_SIZE
59 * - #STORAGE_MAX_BLOCK_SIZE_IN_SAMPLES
60 * - #STORAGE_BLOCK_HEADER_SIZE
61 * - #STORAGE_MAX_UNCOMPRESSED_BLOCK_SIZE_IN_BITS
62 * - #STORAGE_MAX_LOSS_AFTER_CORRUPTION
63 * - #STORAGE_UNCOMPRESSED_BLOCK_SIZE_IN_BITS
64 * - #STORAGE_UNCOMPRESSED_BLOCK_SIZE_IN_BYTES
65 *
66 * @par Choosing correct values
67 * The storage module has many diversity settings, and can be tweaked a lot. Storing data reliably and as
68 * efficient as possible is a complex operation, and every use case will need to adjust a few settings, not
69 * relying on the default values.
70 *
71 * To start, determine the correct sizes of the three memory regions the storage module has exclusive access of:
72 * - General purpose registers: determined via #STORAGE_FIRST_ALON_REGISTER
73 * More general purpose registers means more data can be cached before writing to EEPROM. The more are
74 * assigned, the less EPROM flushes are performed, which prolongs life for batteries with a higher battery
75 * impedance (think printed batteries). But when things fail, all these samples will be lost.
76 * - EEPROM: #STORAGE_EEPROM_FIRST_ROW and #STORAGE_EEPROM_LAST_ROW
77 * By default, a conservative choice is made, leaving ~2 kB alone, for other uses of the application.
78 * The larger the assigned region, the more samples can be stored. This means less room left for the
79 * application to store its state, events and validation results.
80 * - FLASH: #STORAGE_FLASH_FIRST_PAGE and #STORAGE_FLASH_LAST_PAGE
81 * By default, all space not used by the program itself, is used for data storage by the storage module.
82 * Unless your application requires to write to flash for other purposes, it is advised to stick with the
83 * default settings.
84 *
85 * Next, determine what will be stored. Each sample given to the storage module must have the same type. Both
86 * basic types and structures or unions are possible. Each sample will be placed back to back, without a single
87 * padding bit; if you know that a few MSBits are always 0, you can choose to store fewer bits, resulting in a
88 * larger storage capacity.
89 * - #STORAGE_TYPE
90 * - #STORAGE_BITSIZE
91 * - #STORAGE_SIGNED
92 *
93 * Compression greatly increases the number of samples that can be stored, at the cost of the size of the
94 * compression library itself. Normally, the compression ratio is greater when more samples are compressed at
95 * once.
96 * The storage module allows you to insert function calls to compress and decompress a large number of
97 * samples automatically. @n
98 * Keep in mind that enlarging #STORAGE_UNCOMPRESSED_BLOCK_SIZE_IN_BYTES also enlarges
99 * #STORAGE_WORKAREA_SIZE, and the memory required for compressing and decompressing.
100 * - #STORAGE_BLOCK_SIZE_IN_SAMPLES
101 * - #STORAGE_COMPRESS_CB
102 * - #STORAGE_DECOMPRESS_CB
103 * If a compression is to be used which operates on a single sample at a time, the application can wrap the calls
104 * to the storage module and implement this outside the storage module. An example of this is a mapping operation
105 * which reduces the resolution in specific value ranges.
106 * Be sure to adapt the type and bitsize to match the encoded values.
107 *
108 * Data recovery is paramount. Especially when at the end of the use case lifetime, when dealing with batteries
109 * which can't maintain a stable voltage any more, the risk of data corruption increases. The chance for data
110 * corruption to occur is highly dependent on your specific application, your choice of battery, and your
111 * environmental conditions. And apart from data corruption, the IC might also reset during flushing of EEPROM data,
112 * or during a FLASH program operation. When any of this happens, the storage module must be able to return as
113 * many samples as possible.
114 * @n The storage module ensures only a few most recent samples may get lost.
115 * - The samples stored in the general purposes registers were already mentioned. The number can be defined exactly
116 * using #STORAGE_SAMPLE_ALON_CACHE_COUNT
117 * - If data corruption occurs in EEPROM - or when data corruption in EEPROM is suspected - none of the data in
118 * the affected row can be used anymore. The storage module therefore stores a copy the last data row on a
119 * fixed location.
120 * Since the write endurance limit per EEPROM row is only 10.000, the storage module cannot copy this each
121 * time. Via the setting #STORAGE_WRITE_RECOVERY_EVERY_X_SAMPLES the application can decide, taking into
122 * account the maximum number of samples that are to be stored.
123 * If in your use case less than 30.000 samples are stored over the full lifetime of the product, this value
124 * would then be set to 3.
125 * - If data corruption occurs in FLASH, the program will avoid all FLASH operations thereafter. No samples are
126 * lost, but the total storage capacity diminishes, as the FLASH page where the corruption occurred and all
127 * higher FLASH sectors will not be used for data storage any more.
128 * - Writing recovery information in itself can also have an averse impact on the battery. The application is
129 * therefore given an additional choice: the number of EEPROM flushes due to writing recovery data can be
130 * halved, at the expense of an additional 8 bytes of samples which may get lost when data corruption occurs.
131 * If #STORAGE_REDUCE_RECOVERY_WRITES is enabled, only one row is reserved for recovery information, and only
132 * 56 bytes are duplicated.
133 * The maximum amount of samples that may get lost due to an unexpected reset, a battery failure, or a data
134 * corruption, is captured in #STORAGE_MAX_LOSS_AFTER_CORRUPTION
135 * To recover all samples safe for the very last, at the expense of (a lot) more EEPROM flush operations, use these
136 * settings:
137 * @code
138 * #define STORAGE_SAMPLE_ALON_CACHE_COUNT 0
139 * #define STORAGE_WRITE_RECOVERY_EVERY_X_SAMPLES 1
140 * #define STORAGE_REDUCE_RECOVERY_WRITES 0
141 * @endcode
142 * @{
143 */

160#ifndef STORAGE_FIRST_ALON_REGISTER
161 /**
162 * The storage module requires the use of at least one general purpose register. This is used for its own
163 * housekeeping and allows it to finish initialization (#Storage_Init) fast.
164 * In addition, @b all other general purposes registers with an index @b higher @b than given are also reserved for
165 * the storage module. These will be used to minimize EEPROM flush operations by caching as many samples as possible
166 * before committing them to EEPROM.
167 * @pre These registers must be guaranteed not to be touched outside the storage module.
168 * @note More information about the ALON registers can be found in the Power Management Unit driver.
169 * @see STORAGE_SAMPLE_ALON_CACHE_COUNT
170 */
171 #define STORAGE_FIRST_ALON_REGISTER 4
172#endif

179#ifndef STORAGE_EEPROM_FIRST_ROW
180 /**
181 * The first EEPROM row assigned for sample storage. Starting from the first byte in this row, until the last byte
182 * in #STORAGE_EEPROM_LAST_ROW, the storage module has full control: no other code may touch this EEPROM region.
183 * @note By default, the EEPROM row 2 kB below the locked EEPROM rows will be chosen.
184 */
185 #define STORAGE_EEPROM_FIRST_ROW (EEPROM_NR_OF_RW_ROWS - (2048 / EEPROM_ROW_SIZE))
186#endif
187#if !(STORAGE_EEPROM_FIRST_ROW >= 0) || !(STORAGE_EEPROM_FIRST_ROW < EEPROM_NR_OF_RW_ROWS)
188 #error Invalid value for STORAGE_EEPROM_FIRST_ROW
189#endif
190
191#ifndef STORAGE_EEPROM_LAST_ROW
192 /**
193 * The last EEPROM row assigned for sample storage. Starting from the first byte in #STORAGE_EEPROM_FIRST_ROW,
194 * until the last byte in this row, the storage module has full control: no other code may touch this EEPROM region.
195 * @note the size of the EEPROM region is not required to be a multiple of the FLASH sector size.
196 */
197 #define STORAGE_EEPROM_LAST_ROW (EEPROM_NR_OF_RW_ROWS - 1)
198#endif
199#if !(STORAGE_EEPROM_LAST_ROW >= STORAGE_EEPROM_FIRST_ROW) || !(STORAGE_EEPROM_LAST_ROW < EEPROM_NR_OF_RW_ROWS)
200 #error Invalid value for STORAGE_EEPROM_LAST_ROW
201#endif
202
203/** The number of EEPROM rows assigned for sample storage. */
204#define STORAGE_EEPROM_ROW_COUNT (STORAGE_EEPROM_LAST_ROW - STORAGE_EEPROM_FIRST_ROW + 1)
205#if STORAGE_EEPROM_ROW_COUNT < 3
206 #error STORAGE_EEPROM_LAST_ROW and STORAGE_EEPROM_FIRST_ROW must be farther apart: at least 3 EEPROM pages must be assigned for storage
207#endif

209/** The size of the assigned EEPROM storage in bytes */
210#define STORAGE_EEPROM_SIZE (STORAGE_EEPROM_ROW_COUNT * EEPROM_ROW_SIZE)

214#ifndef STORAGE_FLASH_FIRST_PAGE
215 /**
216 * The first FLASH page assigned for storage. Starting from the first byte in this page, until the last byte
217 * in #STORAGE_FLASH_LAST_PAGE, the Storage module has full control: no other code will touch this FLASH region.
218 * @note Two special values exist:
219 * - When equal to @c 0, the first empty FLASH page @b after the @c .text and @c .data
220 * sections will be used. This page is automatically determined during link time by the Storage module's code.
221 * - When a value strict higher than #STORAGE_FLASH_LAST_PAGE is provided, storage to FLASH is disabled. Only the
222 * EEPROM will be used to store the data.
223 * @note The page is not required to be sector aligned.
224 * @warning If a non-zero value is specified, it is the responsibility of the application programmer to ensure the
225 * FLASH location points to outside the @c .text and @c .data sections.
226 */
227 #define STORAGE_FLASH_FIRST_PAGE 0
228#endif
229#if !(STORAGE_FLASH_FIRST_PAGE >= 0) || (STORAGE_FLASH_FIRST_PAGE >= FLASH_NR_OF_RW_SECTORS * FLASH_PAGES_PER_SECTOR)
230 #error Invalid value for STORAGE_FLASH_FIRST_PAGE
231#endif
232
233#ifndef STORAGE_FLASH_LAST_PAGE
234 /**
235 * The last FLASH page assigned for storage. Starting from the first byte in #STORAGE_FLASH_FIRST_PAGE,
236 * until the last byte in this page, the Storage module has full control: no other code will touch this FLASH
237 * region.
238 * @note the size of the region is not required to be a multiple of the FLASH sector size.
239 */
240 #define STORAGE_FLASH_LAST_PAGE (FLASH_NR_OF_RW_SECTORS * FLASH_PAGES_PER_SECTOR - 1)
241#endif
242#if STORAGE_FLASH_LAST_PAGE >= FLASH_NR_OF_RW_SECTORS * FLASH_PAGES_PER_SECTOR
243 #error Invalid value for STORAGE_FLASH_LAST_PAGE
244#endif
245#if STORAGE_FLASH_LAST_PAGE < STORAGE_FLASH_FIRST_PAGE
246 #warning Flash will not be used for data storage
247#endif
248#if (STORAGE_FLASH_FIRST_PAGE == 0) && (STORAGE_FLASH_LAST_PAGE != (FLASH_NR_OF_RW_SECTORS * FLASH_PAGES_PER_SECTOR - 1))
249 #error STORAGE_FLASH_FIRST_PAGE is determined at link time; STORAGE_FLASH_LAST_PAGE can thus not be checked at precompilation time.
250 #error STORAGE_FLASH_LAST_PAGE must be set to the highest possible value in this case.
251#endif

310#ifndef STORAGE_WRITE_RECOVERY_EVERY_X_SAMPLES
311 /**
312 * Update the recovery information after adding @c X samples to @b non-volatile memory. This recovery information
313 * is written on a fixed location. When an unexpected reset occurs or a corruption of the data while writing new
314 * content, the recovery information is used to recover as much data as possible.
315 * @note The recovery information is only written in the call to #Storage_DeInit. Depending on the number of calls
316 * to #Storage_Write and its arguments, it is possible more samples have been written than this number.
317 * @pre maximum number of expected data samples / @c STORAGE_WRITE_RECOVERY_EVERY_X_SAMPLES < 10.000
318 */
319 #define STORAGE_WRITE_RECOVERY_EVERY_X_SAMPLES (1 + STORAGE_SAMPLE_ALON_CACHE_COUNT)
320#endif
321#if STORAGE_WRITE_RECOVERY_EVERY_X_SAMPLES <= 0
322 #error STORAGE_WRITE_RECOVERY_EVERY_X_SAMPLES must be a strict positive number
323#endif
324#if STORAGE_WRITE_RECOVERY_EVERY_X_SAMPLES < STORAGE_SAMPLE_ALON_CACHE_COUNT
325 #warning STORAGE_WRITE_RECOVERY_EVERY_X_SAMPLES is smaller than STORAGE_SAMPLE_ALON_CACHE_COUNT, which makes little sense.
326 #warning Recovery information will be written in Storage_DeInit after STORAGE_SAMPLE_ALON_CACHE_COUNT samples.
327#endif

329#ifndef STORAGE_REDUCE_RECOVERY_WRITES
330 /**
331 * - If not defined, or defined to zero, two rows of the assigned EEPROM region for data storage are used for data
332 * recovery. When data corruption occurs, no extra loss of bytes will occur, only
333 * #STORAGE_SAMPLE_ALON_CACHE_COUNT and #STORAGE_WRITE_RECOVERY_EVERY_X_SAMPLES contribute to this.
334 * - If defined to a non-zero value:
335 * the space used for recovery - where the duplicate data of the last row and the hint information is stored - is
336 * reduced to just one row. When data corruption occurs, an additional loss of up to 8 bytes can occur.
337 */
338 #define STORAGE_REDUCE_RECOVERY_WRITES 0
339#endif
340#if (STORAGE_REDUCE_RECOVERY_WRITES != 0) && (STORAGE_REDUCE_RECOVERY_WRITES != 1)
341 #error Leave STORAGE_REDUCE_RECOVERY_WRITES undefined in app_sel.h, or define it to 0 (--> 2 recovery rows) or 1 (--> 1 recovery row)
342#endif

344/**
345 * The maximum loss of samples that can occur. The storage module may not be able to return the most recently stored
346 * samples after an EEPROM corruption occurs. A scenario where this can happen is when printed batteries are used,
347 * and the battery has an internal impedance of 2 kOhm or higher, resulting in a barely sufficient voltage when
348 * operating normally, and a below-spec voltage when extra load is generated due to an EEPROM flush operation.
349 * @see STORAGE_SAMPLE_ALON_CACHE_COUNT
350 * @see STORAGE_WRITE_RECOVERY_EVERY_X_SAMPLES
351 * @see STORAGE_REDUCE_RECOVERY_WRITES
352 */
353#define STORAGE_MAX_LOSS_AFTER_CORRUPTION 1 \
354 + STORAGE_SAMPLE_ALON_CACHE_COUNT \
355 + STORAGE_WRITE_RECOVERY_EVERY_X_SAMPLES \
356 + STORAGE_REDUCE_RECOVERY_WRITES * STORAGE_IDIVUP(8 * 8, STORAGE_BITSIZE)
357
358/**
359 * The maximum allowed value for #STORAGE_BLOCK_SIZE_IN_SAMPLES. After this many samples, the assigned EEPROM region is
360 * completely filled and no new sample can be written to EEPROM before its contents are moved to FLASH first.
361 * @hideinitializer
362 */
363#if STORAGE_REDUCE_RECOVERY_WRITES
364 #define STORAGE_MAX_BLOCK_SIZE_IN_SAMPLES (((STORAGE_EEPROM_SIZE - 76) * 8) / STORAGE_BITSIZE)
365 /* Magic value 76 is checked at compile time in storage.c */
366#else
367 #define STORAGE_MAX_BLOCK_SIZE_IN_SAMPLES (((STORAGE_EEPROM_SIZE - 140) * 8) / STORAGE_BITSIZE)
368 /* Magic value 140 is checked at compile time in storage.c */
369#endif

379#ifndef STORAGE_BLOCK_SIZE_IN_SAMPLES
380 /**
381 * After writing this number of samples to EEPROM, the module will try to compress them all at once - see
382 * #STORAGE_COMPRESS_CB - and move them to FLASH - defined by #STORAGE_FLASH_FIRST_PAGE and #STORAGE_FLASH_LAST_PAGE.
383 * @note This size determines the value of #STORAGE_WORKAREA_SIZE - the larger this number, the more SRAM is
384 * required, and the larger chunks the compression and decompression algorithms have to work with.
385 * @note By default, the block size will be chosen such that an uncompressed block can fit in one FLASH sector,
386 * including the accompanying meta data.
387 */
388 #define STORAGE_BLOCK_SIZE_IN_SAMPLES (((1024 - STORAGE_BLOCK_HEADER_SIZE) * 8) / STORAGE_BITSIZE)
389 #if STORAGE_BLOCK_SIZE_IN_SAMPLES > STORAGE_MAX_BLOCK_SIZE_IN_SAMPLES
390 #undef STORAGE_BLOCK_SIZE_IN_SAMPLES
391 #define STORAGE_BLOCK_SIZE_IN_SAMPLES STORAGE_MAX_BLOCK_SIZE_IN_SAMPLES
392 #endif
393#endif
394#if ((STORAGE_BLOCK_SIZE_IN_SAMPLES <= 1) || (STORAGE_BLOCK_SIZE_IN_SAMPLES > STORAGE_MAX_BLOCK_SIZE_IN_SAMPLES))
395 #error Invalid value for STORAGE_BLOCK_SIZE_IN_SAMPLES
396#endif

398/** Defines the number of bits required to store one block of samples. */
399#define STORAGE_UNCOMPRESSED_BLOCK_SIZE_IN_BITS (STORAGE_BLOCK_SIZE_IN_SAMPLES * STORAGE_BITSIZE)
